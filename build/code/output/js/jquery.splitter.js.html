  11.  11  /**
  12.  12   * The splitter() plugin implements a two-pane resizable splitter window.
  13.  13   * The selected elements in the jQuery object are converted to a splitter;
  14.  14   * each selected element should have two child elements, used for the panes
  15.  15   * of the splitter. The plugin adds a third child element for the splitbar.
  16.  16   *
  17.  17   * For more details see: http://methvin.com/splitter/
  18.  18   *
  19.  19   *
  20.  20   * @example $('#MySplitter').splitter();
  21.  21   * @desc Create a vertical splitter with default settings
  22.  22   *
  23.  23   * @example $('#MySplitter').splitter({type: 'h', accessKey: 'M'});
  24.  24   * @desc Create a horizontal splitter resizable via Alt+Shift+M
  25.  25   *
  26.  26   * @name splitter
  27.  27   * @type jQuery
  28.  28   * @param Object options Options for the splitter (not required)
  29.  29   * @cat Plugins/Splitter
  30.  30   * @return jQuery
  31.  31   * @author Dave Methvin (dave.methvin@gmail.com)
  32.  32   */
  33.  33  ;
  34.  34  (function($) {
  35.  35  
  36.  36   $.fn.splitter = function(args) {
  37.  37   args = args || {};
  38.  38   return this.each(function() {
  39.  39   var zombie; // left-behind splitbar for outline resizes
  40.  40   function startSplitMouse(evt) {
  41.  41   if (opts.outline)
  42.  42   zombie = zombie || bar.clone(false).insertAfter(A);
  43.  43   panes.css("-webkit-user-select", "none"); // Safari selects A/B text on a move
  44.  44   bar.addClass(opts.activeClass);
  45.  45   $('<div class="splitterMask"></div>').insertAfter(bar);
  46.  46   A._posSplit = A[0][opts.pxSplit] - evt[opts.eventPos];
  47.  47   $(document)
  48.  48   .bind("mousemove", doSplitMouse)
  49.  49   .bind("mouseup", endSplitMouse);
  50.  50   }
  51.  51  
  52.  52   function doSplitMouse(evt) {
  53.  53   var newPos = A._posSplit + evt[opts.eventPos];
  54.  54   if (opts.outline) {
  55.  55   newPos = Math.max(0, Math.min(newPos, splitter._DA - bar._DA));
  56.  56   bar.css(opts.origin, newPos);
  57.  57   } else
  58.  58   resplit(newPos);
  59.  59   }
  60.  60  
  61.  61   function endSplitMouse(evt) {
  62.  62   $('div.splitterMask').remove();
  63.  63   bar.removeClass(opts.activeClass);
  64.  64   var newPos = A._posSplit + evt[opts.eventPos];
  65.  65   if (opts.outline) {
  66.  66   zombie.remove();
  67.  67   zombie = null;
  68.  68   resplit(newPos);
  69.  69   }
  70.  70   panes.css("-webkit-user-select", "text"); // let Safari select text again
  71.  71   $(document)
  72.  72   .unbind("mousemove", doSplitMouse)
  73.  73   .unbind("mouseup", endSplitMouse);
  74.  74   }
  75.  75  
  76.  76   function resplit(newPos) {
  77.  77   // Constrain new splitbar position to fit pane size limits
  78.  78   newPos = Math.max(A._min, splitter._DA - B._max,
  79.  79   Math.min(newPos, A._max, splitter._DA - bar._DA - B._min));
  80.  80   // Resize/position the two panes
  81.  81   bar._DA = bar[0][opts.pxSplit]; // bar size may change during dock
  82.  82   bar.css(opts.origin, newPos).css(opts.fixed, splitter._DF);
  83.  83   A.css(opts.origin, 0).css(opts.split, newPos).css(opts.fixed, splitter._DF);
  84.  84   B.css(opts.origin, newPos + bar._DA)
  85.  85   .css(opts.split, splitter._DA - bar._DA - newPos).css(opts.fixed, splitter._DF);
  86.  86   // IE fires resize for us; all others pay cash
  87.  87   if (!$.browser.msie)
  88.  88   panes.trigger("resize");
  89.  89   }
  90.  90  
  91.  91   function dimSum(jq, dims) {
  92.  92   // Opera returns -1 for missing min/max width, turn into 0
  93.  93   var sum = 0;
  94.  94   for (var i = 1; i < arguments.length; i++)
  95.  95   sum += Math.max(parseInt(jq.css(arguments[i])) || 0, 0);
  96.  96   return sum;
  97.  97   }
  98.  98  
  99.  99   // Determine settings based on incoming opts, element classes, and defaults
  100. 100   var vh = (args.splitHorizontal ? 'h' : args.splitVertical ? 'v' : args.type) || 'v';
  101. 101   var opts = $.extend({
  102. 102   activeClass: 'active', // class name for active splitter
  103. 103   pxPerKey: 8, // splitter px moved per keypress
  104. 104   tabIndex: 0, // tab order indicator
  105. 105   accessKey: '' // accessKey for splitbar
  106. 106   }, {
  107. 107   v: { // Vertical splitters:
  108. 108   keyLeft: 39, keyRight: 37, cursor: "e-resize",
  109. 109   splitbarClass: "vsplitbar", outlineClass: "voutline",
  110. 110   type: 'v', eventPos: "pageX", origin: "left",
  111. 111   split: "width", pxSplit: "offsetWidth", side1: "Left", side2: "Right",
  112. 112   fixed: "height", pxFixed: "offsetHeight", side3: "Top", side4: "Bottom"
  113. 113   },
  114. 114   h: { // Horizontal splitters:
  115. 115   keyTop: 40, keyBottom: 38, cursor: "n-resize",
  116. 116   splitbarClass: "hsplitbar", outlineClass: "houtline",
  117. 117   type: 'h', eventPos: "pageY", origin: "top",
  118. 118   split: "height", pxSplit: "offsetHeight", side1: "Top", side2: "Bottom",
  119. 119   fixed: "width", pxFixed: "offsetWidth", side3: "Left", side4: "Right"
  120. 120   }
  121. 121   }[vh], args);
  122. 122  
  123. 123   // Create jQuery object closures for splitter and both panes
  124. 124   var splitter = $(this).css({position: "relative"});
  125. 125   var panes = $(">*", splitter[0]).css({
  126. 126   position: "absolute", // positioned inside splitter container
  127. 127   "z-index": "1", // splitbar is positioned above
  128. 128   "-moz-outline-style": "none" // don't show dotted outline
  129. 129   });
  130. 130   var A = $(panes[0]); // left or top
  131. 131   var B = $(panes[1]); // right or bottom
  132. 132  
  133. 133   // Focuser element, provides keyboard support; title is shown by Opera accessKeys
  134. 134   var focuser = $('<a href="javascript:void(0)"></a>')
  135. 135   .attr({accessKey: opts.accessKey, tabIndex: opts.tabIndex, title: opts.splitbarClass})
  136. 136   .bind($.browser.opera ? "click" : "focus", function() {
  137. 137   this.focus();
  138. 138   bar.addClass(opts.activeClass)
  139. 139   })
  140. 140   .bind("keydown", function(e) {
  141. 141   var key = e.which || e.keyCode;
  142. 142   var dir = key == opts["key" + opts.side1] ? 1 : key == opts["key" + opts.side2] ? -1 : 0;
  143. 143   if (dir)
  144. 144   resplit(A[0][opts.pxSplit] + dir * opts.pxPerKey, false);
  145. 145   })
  146. 146   .bind("blur", function() {
  147. 147   bar.removeClass(opts.activeClass)
  148. 148   });
  149. 149  
  150. 150   // Splitbar element, can be already in the doc or we create one
  151. 151   var bar = $(panes[2] || '<div></div>')
  152. 152   .insertAfter(A).css("z-index", "100").append(focuser)
  153. 153   .attr({"class": opts.splitbarClass, unselectable: "on"})
  154. 154   .css({position: "absolute", "user-select": "none", "-webkit-user-select": "none",
  155. 155   "-khtml-user-select": "none", "-moz-user-select": "none", "top": "0px"})
  156. 156   .bind("mousedown", startSplitMouse);
  157. 157   // Use our cursor unless the style specifies a non-default cursor
  158. 158   if (/^(auto|default|)$/.test(bar.css("cursor")))
  159. 159   bar.css("cursor", opts.cursor);
  160. 160  
  161. 161   // Cache several dimensions for speed, rather than re-querying constantly
  162. 162   bar._DA = bar[0][opts.pxSplit];
  163. 163   splitter._PBF = $.boxModel ? dimSum(splitter, "border" + opts.side3 + "Width", "border" + opts.side4 + "Width") : 0;
  164. 164   splitter._PBA = $.boxModel ? dimSum(splitter, "border" + opts.side1 + "Width", "border" + opts.side2 + "Width") : 0;
  165. 165   A._pane = opts.side1;
  166. 166   B._pane = opts.side2;
  167. 167   $.each([A,B], function() {
  168. 168   this._min = opts["min" + this._pane] || dimSum(this, "min-" + opts.split);
  169. 169   this._max = opts["max" + this._pane] || dimSum(this, "max-" + opts.split) || 9999;
  170. 170   this._init = opts["size" + this._pane] === true ?
  171. 171   parseInt($.curCSS(this[0], opts.split)) : opts["size" + this._pane];
  172. 172   });
  173. 173  
  174. 174   // Determine initial position, get from cookie if specified
  175. 175   var initPos = A._init;
  176. 176   if (!isNaN(B._init)) // recalc initial B size as an offset from the top or left side
  177. 177   initPos = splitter[0][opts.pxSplit] - splitter._PBA - B._init - bar._DA;
  178. 178   if (opts.cookie) {
  179. 179   if (!$.cookie)
  180. 180   alert('jQuery.splitter(): jQuery cookie plugin required');
  181. 181   var ckpos = parseInt($.cookie(opts.cookie));
  182. 182   if (!isNaN(ckpos))
  183. 183   initPos = ckpos;
  184. 184   $(window).bind("unload", function() {
  185. 185   var state = String(bar.css(opts.origin)); // current location of splitbar
  186. 186   $.cookie(opts.cookie, state, {expires: opts.cookieExpires || 365,
  187. 187   path: opts.cookiePath || document.location.pathname});
  188. 188   });
  189. 189   }
  190. 190   if (isNaN(initPos)) // King Solomon's algorithm
  191. 191   initPos = Math.round((splitter[0][opts.pxSplit] - splitter._PBA - bar._DA) / 2);
  192. 192  
  193. 193   // Resize event propagation and splitter sizing
  194. 194   if (opts.anchorToWindow) {
  195. 195   // Account for margin or border on the splitter container and enforce min height
  196. 196   splitter._hadjust = dimSum(splitter, "borderTopWidth", "borderBottomWidth", "marginBottom");
  197. 197   splitter._hmin = Math.max(dimSum(splitter, "minHeight"), 20);
  198. 198   $(window).bind("resize",
  199. 199   function() {
  200. 200   var top = splitter.offset().top;
  201. 201   var wh = $(window).height();
  202. 202   splitter.css("height", Math.max(wh - top - splitter._hadjust, splitter._hmin) + "px");
  203. 203   if (!$.browser.msie) splitter.trigger("resize");
  204. 204   }).trigger("resize");
  205. 205   }
  206. 206   else if (opts.resizeToWidth && !$.browser.msie)
  207. 207   $(window).bind("resize", function() {
  208. 208   splitter.trigger("resize");
  209. 209   });
  210. 210  
  211. 211   // Resize event handler; triggered immediately to set initial position
  212. 212   splitter.bind("resize",
  213. 213   function(e, size) {
  214. 214   // Custom events bubble in jQuery 1.3; don't Yo Dawg
  215. 215   if (e.target != this) return;
  216. 216   // Determine new width/height of splitter container
  217. 217   splitter._DF = splitter[0][opts.pxFixed] - splitter._PBF;
  218. 218   splitter._DA = splitter[0][opts.pxSplit] - splitter._PBA;
  219. 219   // Bail if splitter isn't visible or content isn't there yet
  220. 220   if (splitter._DF <= 0 || splitter._DA <= 0) return;
  221. 221   // Re-divvy the adjustable dimension; maintain size of the preferred pane
  222. 222   resplit(!isNaN(size) ? size : (!(opts.sizeRight || opts.sizeBottom) ? A[0][opts.pxSplit] :
  223. 223   splitter._DA - B[0][opts.pxSplit] - bar._DA));
  224. 224   }).trigger("resize", [initPos]);
  225. 225   });
  226. 226   };
